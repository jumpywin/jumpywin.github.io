```sql
create schema if not exists luoxiaohei collate utf8mb4_0900_ai_ci;

create table if not exists ability
(
	id int auto_increment
		primary key,
	name char(20) not null,
	note text null
);

create table if not exists racial
(
	id int auto_increment
		primary key,
	name char(20) not null
);

create table if not exists `character`
(
	id int auto_increment
		primary key,
	name char(20) not null,
	racial int null,
	note text null,
	constraint fk_racial
		foreign key (racial) references racial (id)
			on update cascade
);

create table if not exists `character-ability`
(
	character_id int not null,
	ability_id int null,
	constraint fk_ability
		foreign key (ability_id) references ability (id)
			on update cascade,
	constraint fk_character
		foreign key (character_id) references `character` (id)
			on update cascade
);
```
